#
# 3. Install requirements:
#    pip install -r requirements.txt
#

#    pip install torch==2.9.0+cu121 torchvision==0.24.0+cu121 --index-url https://download.pytorch.org/whl/cu121

#    - CLERK_SECRET_KEY: Clerk authentication secret
